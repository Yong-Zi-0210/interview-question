```js
  async function streamChat(prompt: string, onChunk: (text: string) => void) {
  const response = await fetch('/api/chat', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ prompt }),
  });

  const reader = response.body!.getReader();
  const decoder = new TextDecoder();

  while (true) {
    const { done, value } = await reader.read();
    if (done) break;

    const chunk = decoder.decode(value, { stream: true });
    // SSE 格式: data: {"content": "xxx"}\n\n
    const lines = chunk.split('\n').filter(line => line.startsWith('data:'));

    for (const line of lines) {
      const json = JSON.parse(line.slice(5));
      onChunk(json.content);
    }
  }
}
```

function useTypewriter() {
  const displayText = ref('');
  const queue: string[] = [];
  let isTyping = false;

  function addText(text: string) {
    queue.push(...text.split(''));
    if (!isTyping) startTyping();
  }

  function startTyping() {
    isTyping = true;
    const timer = setInterval(() => {
      if (queue.length === 0) {
        isTyping = false;
        clearInterval(timer);
        return;
      }
      displayText.value += queue.shift();
    }, 30);
  }

  return { displayText, addText };
}

```js
interface Message {
  role: 'user' | 'assistant';
  content: string;
}

function truncateContext(messages: Message[], maxTokens: number): Message[] {
  // 简单策略：保留系统提示 + 最近的 N 条消息
  const systemMsg = messages.find(m => m.role === 'system');
  const conversationMsgs = messages.filter(m => m.role !== 'system');

  let totalTokens = estimateTokens(systemMsg?.content || '');
  const result: Message[] = systemMsg ? [systemMsg] : [];

  // 从后往前添加，保留最近的对话
  for (let i = conversationMsgs.length - 1; i >= 0; i--) {
    const msg = conversationMsgs[i];
    const msgTokens = estimateTokens(msg.content);

    if (totalTokens + msgTokens > maxTokens) break;

    result.unshift(msg);
    totalTokens += msgTokens;
  }

  return result;
}

function estimateTokens(text: string): number {
  // 粗略估计：中文 1 字 ≈ 2 tokens，英文 1 词 ≈ 1 token
  return Math.ceil(text.length * 1.5);
}
```

interface ComponentSchema {
  type: string;           // 组件类型
  props: Record<string, any>;  // 组件属性
  children?: ComponentSchema[]; // 子组件
  events?: Record<string, string>; // 事件绑定
}

// 示例 Schema
const pageSchema: ComponentSchema = {
  type: 'Page',
  props: { title: '用户管理' },
  children: [
    {
      type: 'Form',
      props: { labelWidth: '100px' },
      children: [
        {
          type: 'Input',
          props: { label: '用户名', model: 'form.username' },
        },
        {
          type: 'Select',
          props: { label: '角色', model: 'form.role', options: [] },
        },
      ],
    },
  ],
};

// Schema → Vue 组件递归渲染
function renderSchema(schema: ComponentSchema) {
  const Component = resolveComponent(schema.type);

  return h(
    Component,
    schema.props,
    schema.children?.map(child => renderSchema(child))
  );
}

import { useDraggable, useDroppable } from '@vueuse/core';

function useDragDrop(components: Ref<ComponentSchema[]>) {
  const dragIndex = ref(-1);
  const dropIndex = ref(-1);

  function onDragStart(index: number) {
    dragIndex.value = index;
  }

  function onDrop(targetIndex: number) {
    if (dragIndex.value === -1) return;

    const [removed] = components.value.splice(dragIndex.value, 1);
    components.value.splice(targetIndex, 0, removed);

    dragIndex.value = -1;
  }

  return { onDragStart, onDrop };
}

function generateVueCode(schema: ComponentSchema): string {
  const template = generateTemplate(schema);
  const script = generateScript(schema);

  return `
<template>
${template}
</template>

<script setup lang="ts">
${script}
</script>
`;
}

function generateTemplate(schema: ComponentSchema, indent = 2): string {
  const spaces = ' '.repeat(indent);
  const tag = getTagName(schema.type);
  const props = Object.entries(schema.props)
    .map(([k, v]) => `:${k}="${v}"`)
    .join(' ');

  if (!schema.children?.length) {
    return `${spaces}<${tag} ${props} />`;
  }

  const children = schema.children
    .map(child => generateTemplate(child, indent + 2))
    .join('\n');

  return `${spaces}<${tag} ${props}>\n${children}\n${spaces}</${tag}>`;
}
